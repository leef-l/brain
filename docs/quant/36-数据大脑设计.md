## 一、职责定义

```yaml
brain:
  kind: data
  version: "1.0.0"
  capabilities:
    - market_data.realtime
    - market_data.historical
    - feature.compute
    - feature.vector
    - orderbook.snapshot
    - microstructure.metrics
    - data.validation
    - data.replay
```

Data Brain **不是** OKX 客户端。它是一个统一数据中枢，所有数据源通过插件接口接入。

```go
    for _, cfg := range configs {
        factory, ok := r.factories[cfg.Type]
        if !ok {

        r.providers[cfg.Name] = provider
    }
    return nil
```

## 五、数据质量校验层（DataValidator）

```go
type DataValidator struct {
    lastPrice    map[string]float64
    lastTS       map[string]int64
    gapDetector  *GapDetector
    alertSink    func(alert ValidatorAlert)
}

type ValidatorAlert struct {
    Level   string    // "warning", "critical"
    Type    string    // "price_spike", "gap", "stale", "inconsistent"
    Symbol  string
    Detail  string
    EventTS time.Time
}

func (v *DataValidator) Validate(event *DataEvent) bool {
    switch event.Category {
    case CategoryMarket:
        return v.validateMarket(event)
    default:
        return true
    }
}

func (v *DataValidator) validateMarket(event *DataEvent) bool {
    now := time.Now().UnixMilli()

    // 1. 时间戳 — 不接受未来数据或过期数据
    if event.Timestamp > now+5000 {
        v.alert("warning", "future_timestamp", event.Symbol, "...")
        return false
    }
    if event.Timestamp < now-300000 {
        v.alert("warning", "stale_data", event.Symbol, "...")
        return false
    }

    // 2. 价格突变 — 单 tick 变动 >10% 丢弃
    if price := extractPrice(event); price > 0 {
        if last, ok := v.lastPrice[event.Symbol]; ok && last > 0 {
            change := math.Abs(price-last) / last
            if change > 0.10 {
                v.alert("critical", "price_spike", event.Symbol,
                    fmt.Sprintf("%.2f → %.2f (%.1f%%)", last, price, change*100))
                return false
            }
        }
        v.lastPrice[event.Symbol] = price
    }

    // 3. 重复时间戳 — 丢弃
    if last, ok := v.lastTS[event.Symbol]; ok && event.Timestamp <= last {
        return false
    }
    v.lastTS[event.Symbol] = event.Timestamp

    // 4. K线缺失 gap — 连续缺失 >3 根触发 REST 回填
    v.gapDetector.Observe(event)

    return true
}
```

校验规则通过配置调整：

```yaml
validation:
  max_price_change_pct: 10.0
  max_future_ts_ms: 5000
  max_stale_ts_ms: 300000
  gap_backfill_threshold: 3
```

```go
func (p *OKXSwapProvider) onWSMessage(channel, instID string, data []byte) {
    event := DataEvent{
        Provider: p.name,
        Category: CategoryMarket,
        Symbol:   p.normalizer.Normalize("okx", instID),
        LocalTS:  time.Now().UnixNano(),
    }
    switch {
    case strings.HasPrefix(channel, "candle"):
        event.Topic    = fmt.Sprintf("candle.%s.%s", channel[6:], instID)
        event.Priority = PriorityNearRT
        event.Payload  = parseCandlePayload(data)
    case channel == "books5":
        event.Topic    = fmt.Sprintf("orderbook.%s", instID)
        event.Priority = PriorityRealtime
        event.Payload  = parseOrderBookPayload(data)
    case channel == "trades":
        event.Topic    = fmt.Sprintf("trade.%s", instID)
        event.Priority = PriorityRealtime
        event.Payload  = parseTradePayload(data)
    case channel == "funding-rate":
        event.Topic    = fmt.Sprintf("funding.%s", instID)
        event.Priority = PriorityNearRT
        event.Payload  = parseFundingPayload(data)
    }
```

## 七、回测 Provider（ReplayProvider）

```go
type ReplayProvider struct {
    name      string
    store     *storage.Repositories
    speed     float64  // 1.0 = 实时, 0 = 尽速回放
    eventSink DataSink
}

func (p *ReplayProvider) Replay(ctx context.Context, from, to time.Time, speed float64) error {
    candles, err := p.store.Candles().QueryRange(ctx, from, to)
    if err != nil {
        return err
    }
    var prevTS int64
    for _, c := range candles {
        if speed > 0 && prevTS > 0 {
            delay := time.Duration(float64(c.Timestamp-prevTS)*1e6) / time.Duration(speed)
            time.Sleep(delay)
        }
        prevTS = c.Timestamp
        p.eventSink.OnEvent(DataEvent{
            Provider:  p.name,
            Category:  CategoryMarket,
            Symbol:    c.InstID,
            Topic:     fmt.Sprintf("candle.%s.%s", c.Bar, c.InstID),
            Timestamp: c.Timestamp,
            LocalTS:   time.Now().UnixNano(),
            Priority:  PriorityNearRT,
            Payload:   c,
        })
    }
    return nil
}
```

**关键**: Quant Brain 不需要任何修改 — 它只消费 MarketView，不关心数据来自实时还是回放。

## 九、Event Router

```go
type EventRouter struct {
    realtimeCh chan DataEvent  // 无缓冲，实时
    nearRTCh   chan DataEvent  // 缓冲 1024
    batchCh    chan DataEvent  // 缓冲 4096
}

func (r *EventRouter) OnEvent(event DataEvent) {
    switch event.Priority {
    case PriorityRealtime:
        select {
        case r.realtimeCh <- event:
        default:
            // tick 丢一个无影响，metric++
        }
    case PriorityNearRT:
        r.nearRTCh <- event
    case PriorityBatch:
        r.batchCh <- event
    }
}
```

## 十一、特征计算引擎

```go
type FeatureEngine struct {
    candles   *CandleAggregator
    orderbook *OrderBookTracker
    tradeflow *TradeFlowAnalyzer
    enricher  *ContextEnricher

// 特征向量组成（192 维）
//
// [0:60]    价格特征      多时间框架 EMA 比率、RSI、MACD、BB、ATR、ADX
// [60:100]  成交量特征    量比、OBV 斜率、大单占比、成交密度
// [100:130] 微观结构      订单簿不平衡、Spread、Toxicity、资金费率
```

各区段详细计算逻辑：

```go
func (e *FeatureEngine) computeFeatureVector(instID string) []float64 {
    vec := make([]float64, e.vectorDim)

    // --- 价格特征 [0:60] — 每个时间框架 12 维 × 5 TF ---
    for i, tf := range []string{"1m", "5m", "15m", "1H", "4H"} {
        w := e.candles.GetWindow(instID, tf)
        if w == nil { continue }
        base := i * 12
        price := w.Current.Close

        vec[base+0]  = (price - w.ema9.Value()) / price
        vec[base+1]  = (price - w.ema21.Value()) / price
        vec[base+2]  = (price - w.ema55.Value()) / price
        vec[base+3]  = (w.ema9.Value() - w.ema21.Value()) / price
        vec[base+4]  = w.rsi14.Value() / 100.0
        vec[base+5]  = w.macd.Histogram() / price
        vec[base+6]  = w.bbPosition()
        vec[base+7]  = w.atr14.Value() / price
        vec[base+8]  = w.priceChangeRate(5)
        vec[base+9]  = w.priceChangeRate(20)
        vec[base+10] = w.volatility(20)
        vec[base+11] = w.adx.Value() / 100.0
    }

    // --- 成交量 [60:100] ---
    // ... 量比、OBV、大单

    // --- 微观结构 [100:130] ---
    // ... 订单簿、Toxicity

    // --- 时序 [130:160] ---
    // ... 多尺度动量、波动

    // --- 跨品种 [160:180] ---
    // ... BTC/ETH 联动

    // --- 宏观/情绪 [180:192] ---
    // ... 由 ContextEnricher 注入

    return vec
}
```

## 十二、上下文增强（Context Enricher）

```go
type ContextEnricher struct {
    macroSignal   atomic.Value  // *MacroSignal
    newsSignal    atomic.Value  // *NewsSignal

type MacroSignal struct {
    Sentiment    float64

    CPIDelta     *float64
    FedHawkish   float64
    NextEventIn  time.Duration
    Flags        uint64       // IS_PRE_NFP | IS_PRE_FOMC | ...
    UpdatedAt    time.Time

func (e *FeatureEngine) injectContext(snap *MarketSnapshot) {
    if macro := e.enricher.macroSignal.Load(); macro != nil {
        m := macro.(*MacroSignal)
        snap.MacroSentiment = m.Sentiment
        snap.ContextFlags |= m.Flags
    }
    if news := e.enricher.newsSignal.Load(); news != nil {
        snap.NewsSentiment = news.(*NewsSignal).Sentiment
    }
    if chain := e.enricher.chainSignal.Load(); chain != nil {
        snap.ChainFlowScore = chain.(*ChainSignal).FlowScore
    }
}
```

```go
type RingHeader struct {

    SlotCount  uint32  // 1024
    SlotSize   uint32  // 1792
    CreatedAt  int64
    _reserved  [32]byte
}

// 每品种 ≈ 1.75 MiB，3 品种 ≈ 5.25 MiB
```

## 十四、数据存储策略

| 数据类型 | 频率 | 存储目标 | 保留 |
|---------|------|---------|------|
| 1m K线 | 每分钟 | PostgreSQL `candles` | 60d |
| 5m/15m/1H/4H K线 | 按周期 | PostgreSQL `candles` | 永久 |
| 特征向量 | 每分钟 | PostgreSQL `vectors` | 60d/180d |
| 订单簿快照 | 不持久化 | 内存 ring buffer | — |
| 逐笔成交 | 不持久化 | 内存 5m 窗口 | — |
| 宏观数据 | 按事件 | PostgreSQL `macro_events` | 永久 |
| 新闻 | 按事件 | PostgreSQL `news_events` | 90d |
| 链上数据 | 按事件 | PostgreSQL `chain_events` | 30d |
| 质量告警 | 按事件 | PostgreSQL `validator_alerts` | 30d |
