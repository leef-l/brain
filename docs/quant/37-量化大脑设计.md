### 3.3 设计要点

- **策略计算全局共享** — 4 个策略只算一次，信号广播给所有账号
- **风控完全隔离** — 每个账号独立 Guard、独立 MemoryState
- **执行完全隔离** — 每个账号独立 Executor、独立凭据
- **可 per-account 覆盖风控** — 子账号可以更保守
- **可 per-account 过滤策略/品种** — 子账号只跑部分策略

## 四、全局风控层（GlobalRiskGuard）

位于 Aggregator 之后、AccountRouter 之前。跨账号视角：

```go
type GlobalRiskGuard struct {
    accounts     map[string]*AccountState
    config       GlobalRiskConfig
}

type GlobalRiskConfig struct {
    MaxGlobalExposurePct   float64  // 所有账号总敞口/总权益，默认 50%
    MaxGlobalSameDirection float64  // 同方向总敞口上限，默认 30%
    MaxGlobalDailyLoss     float64  // 所有账号累计日亏损上限，默认 5%
    MaxSymbolExposure      float64  // 单品种跨账号总敞口上限，默认 15%
}

func (g *GlobalRiskGuard) Evaluate(signal *strategy.AggregatedSignal) *risk.Decision {
    totalEquity := 0.0
    totalExposure := 0.0
    symbolExposure := 0.0
    sameDirectionExp := 0.0
    dailyLossTotal := 0.0

    for _, acct := range g.accounts {
        if acct.Status != "active" { continue }
        eq := acct.Memory.Equity()
        totalEquity += eq

        for _, pos := range acct.Memory.Positions() {
            notional := math.Abs(pos.Notional)
            totalExposure += notional
            if pos.Symbol == signal.Symbol {
                symbolExposure += notional
            }
            if string(pos.Direction) == string(signal.Direction) {
                sameDirectionExp += notional
            }
        }
        dailyLossTotal += acct.DailyPnL
    }

    if totalEquity == 0 {
        return &risk.Decision{Allowed: true}
    }

    // 检查 4 个全局限制
    if totalExposure/totalEquity > g.config.MaxGlobalExposurePct/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global exposure exceeded"}
    }
    if symbolExposure/totalEquity > g.config.MaxSymbolExposure/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: fmt.Sprintf("%s global exposure exceeded", signal.Symbol)}
    }
    if sameDirectionExp/totalEquity > g.config.MaxGlobalSameDirection/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global same-direction exceeded"}
    }
    if dailyLossTotal/totalEquity < -g.config.MaxGlobalDailyLoss/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global daily loss exceeded"}
    }

    return &risk.Decision{Allowed: true}
}
```

**未来预留**: 如果风控复杂度增长，GlobalRiskGuard 可以零修改拉出来变成独立 Risk Brain sidecar。

## 五、信号审计追踪（SignalTrace）

每笔交易决策的完整链路可追溯：

```go
type SignalTrace struct {
    TraceID    string
    Timestamp  time.Time
    Symbol     string

    // 数据快照
    SnapshotSeq uint64
    Price       float64
    Features    [192]float64

    // 各策略原始信号
    Signals     []strategy.Signal

    // 聚合结果
    Aggregated  strategy.AggregatedSignal

    // 全局风控结果
    GlobalRisk  risk.Decision

    // 各账号处理结果
    AccountResults []AccountTraceResult
}

type AccountTraceResult struct {
    AccountID  string
    RiskResult risk.Decision
    Intent     *execution.OrderIntent
    ExecResult *execution.ExecutionResult
    Latency    time.Duration
}

type TraceStore interface {
    Save(ctx context.Context, trace *SignalTrace) error
    Query(ctx context.Context, filter TraceFilter) ([]SignalTrace, error)
}
```

审计追踪回答这些问题：
- 这笔交易为什么开？→ 哪个 tick 触发？
- 每个策略给了什么信号？→ Signals 数组
- 聚合后的分数是多少？→ Aggregated
- 全局风控通过了吗？→ GlobalRisk
- 每个账号怎么处理的？→ AccountResults
- 下单延迟多少？→ Latency

## 六、Tick Loop（核心循环）

```go
type QuantBrain struct {
    rings       map[string]*SharedRingBuffer
    pool        *strategy.Pool
    aggregator  *strategy.Aggregator
    globalGuard *GlobalRiskGuard
    router      *AccountRouter
    traceStore  TraceStore
    kernelCaller sidecar.KernelCaller
}

func (q *QuantBrain) Run(ctx context.Context) error {
    ticker := time.NewTicker(100 * time.Millisecond)  // 10 Hz
    defer ticker.Stop()
    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-ticker.C:
            q.tick(ctx)
        }
    }
}

func (q *QuantBrain) tick(ctx context.Context) {
    for instID, ring := range q.rings {
        snap, ok := ring.ReadLatest()
        if !ok { continue }

        // 1. 适配 MarketView
        view := q.adaptSnapshot(instID, snap)

        // 2. 策略池（全局，只算一次）
        signals := q.pool.Compute(view)

        // 3. 信号聚合
        agg := q.aggregator.Aggregate(signals, q.buildReviewContext())
        if agg.Direction == strategy.DirectionFlat ||
           agg.Direction == strategy.DirectionHold {
            continue
        }

        // 4. 开始审计追踪
        trace := &SignalTrace{
            TraceID:     generateTraceID(),
            Timestamp:   time.Now(),
            Symbol:      instID,
            SnapshotSeq: snap.SeqNum,
            Price:       snap.CurrentPrice,
            Features:    snap.FeatureVector,
            Signals:     signals,
            Aggregated:  *agg,
        }

        // 5. 全局风控
        globalDecision := q.globalGuard.Evaluate(agg)
        trace.GlobalRisk = *globalDecision
        if !globalDecision.Allowed {
            q.traceStore.Save(ctx, trace)
            continue
        }

        // 6. LLM 审查（条件触发）
        if q.NeedsReview(agg) {
            review := q.RequestReview(ctx, agg, snap)
            if !review.Approved {
                q.traceStore.Save(ctx, trace)
                continue
            }
        }

        // 7. 多账号分发
        results := q.router.Dispatch(ctx, agg, snap)
        trace.AccountResults = results

        // 8. 保存审计追踪
        q.traceStore.Save(ctx, trace)
    }

    // 9. 各账号处理 tick（止损/止盈）
    for _, acct := range q.router.accounts {
        q.processAccountOrders(ctx, acct)
    }
}
```

## 八、Account Router

```go
type AccountRouter struct {
    accounts   map[string]*AccountState
    normalizer *InstrumentNormalizer
}

func (r *AccountRouter) Dispatch(ctx context.Context, signal *strategy.AggregatedSignal, snap *MarketSnapshot) []AccountTraceResult {
    results := make([]AccountTraceResult, 0, len(r.accounts))

    for _, acct := range r.accounts {
        tr := AccountTraceResult{AccountID: acct.Config.ID}
        start := time.Now()

        if !acct.Config.Enabled || acct.Status != "active" { continue }
        if !acct.isStrategyAllowed(signal.DominantStrategy) { continue }
        if !acct.isInstrumentAllowed(signal.Symbol) { continue }

        // per-account 风控
        decision := acct.Guard.Evaluate(acct.buildOrderRequest(signal, snap))
        tr.RiskResult = *decision
        if !decision.Allowed {
            results = append(results, tr)
            continue
        }

        // 仓位计算（应用 WeightFactor）
        intent := acct.buildOrderIntent(signal, snap)
        intent.Quantity = scaleQuantity(intent.Quantity, acct.Config.WeightFactor)
        intent.Symbol = r.normalizer.Denormalize(acct.Config.Exchange, signal.Symbol)
        tr.Intent = &intent

        // 执行
        result, err := acct.Executor.Execute(ctx, intent)
        if err != nil {
            acct.Status = "error"
            acct.LastError = err.Error()
            tr.Latency = time.Since(start)
            results = append(results, tr)
            continue
        }

        acct.Memory.Update(result)
        tr.ExecResult = &result
        tr.Latency = time.Since(start)
        results = append(results, tr)
    }
    return results
}
```

## 十、LLM 审查触发

```go
func (q *QuantBrain) NeedsReview(agg *strategy.AggregatedSignal) bool {
    ctx := q.buildReviewContext()
    return ctx.OpenPositions >= q.config.LLM.TriggerConcurrent ||
           ctx.LargestPositionPct > q.config.LLM.TriggerPositionPct ||
           ctx.DailyLossPct > q.config.LLM.TriggerDailyLoss ||
           ctx.AnomalyDetected
}

func (q *QuantBrain) RequestReview(ctx context.Context, agg *strategy.AggregatedSignal, snap *MarketSnapshot) *ReviewDecision {
    result, err := q.kernelCaller.Call(ctx, "subtask.delegate", &protocol.SubtaskRequest{
        TargetKind:  "central",
        Instruction: "review_trade",
        Context:     marshalJSON(reviewRequest),
        Budget:      &protocol.SubtaskBudget{MaxTokens: 500, TimeoutSec: 10},
    })
    if err != nil || result.Status == "timeout" {
        // LLM 超时不阻塞，降级为自动决策
        return &ReviewDecision{Approved: true, Reason: "llm_timeout_fallback"}
    }
    return parseReviewDecision(result.Output)
}
```

## 十一、崩溃恢复

```go
func (q *QuantBrain) recoverState(ctx context.Context) error {
    for _, acct := range q.router.accounts {
        // 1. 从 PostgreSQL 恢复交易记录 → MemoryState
        trades, err := q.store.Trades().OpenByAccount(ctx, acct.Config.ID)
        if err != nil { return err }
        for _, t := range trades {
            acct.Memory.RestorePosition(t)
        }

        // 2. live 账号重连 Private WS → 获取最新持仓
        if acct.Config.Mode == "live" && acct.PrivateWS != nil {
            acct.PrivateWS.Reconnect(ctx)
            // 用交易所返回的持仓数据校正 MemoryState
        }

        // 3. 恢复今日损益
        acct.DailyPnL = q.store.Trades().DailyPnL(ctx, acct.Config.ID, time.Now())
    }
    return nil
}
```
