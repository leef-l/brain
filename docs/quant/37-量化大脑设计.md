### 3.3 设计要点

- **策略计算全局共享** — 4 个策略只算一次，信号广播给所有账号
- **风控完全隔离** — 每个账号独立 Guard、独立 MemoryState
- **执行完全隔离** — 每个账号独立 Executor、独立凭据
- **可 per-account 覆盖风控** — 子账号可以更保守
- **可 per-account 过滤策略/品种** — 子账号只跑部分策略

## 四、全局风控层（GlobalRiskGuard）

位于“候选单构建”之后、“实际下单”之前。它不直接审 `AggregatedSignal`，而是审本次准备发送到各账号的 `DispatchPlan`：

```go
type DispatchCandidate struct {
    AccountID         string
    Symbol            string
    Direction         strategy.Direction
    ProposedQty       float64
    ProposedNotional  float64
    WeightFactor      float64
    RiskResult        risk.Decision // per-account 预检查结果
}

type DispatchPlan struct {
    TraceID     string
    Signal      *strategy.AggregatedSignal
    Snapshot    *MarketSnapshot
    Candidates  []DispatchCandidate
    Review      *ReviewDecision
}

func (p *DispatchPlan) ApplySizeFactor(factor float64) {
    factor = clamp(factor, 0, 1)
    for i := range p.Candidates {
        p.Candidates[i].ProposedQty *= factor
        p.Candidates[i].ProposedNotional *= factor
    }
}

type GlobalRiskGuard struct {
    accounts     map[string]*AccountState
    config       GlobalRiskConfig
}

type GlobalRiskConfig struct {
    MaxGlobalExposurePct   float64  // 所有账号总敞口/总权益，默认 50%
    MaxGlobalSameDirection float64  // 同方向总敞口上限，默认 30%
    MaxGlobalDailyLoss     float64  // 所有账号累计日亏损上限，默认 5%
    MaxSymbolExposure      float64  // 单品种跨账号总敞口上限，默认 15%
}

func (g *GlobalRiskGuard) Evaluate(plan *DispatchPlan) *risk.Decision {
    totalEquity := 0.0
    existingExposure := 0.0
    proposedExposure := 0.0
    symbolExposure := 0.0
    sameDirectionExp := 0.0
    dailyLossTotal := 0.0

    for _, acct := range g.accounts {
        if acct.Status != "active" { continue }
        eq := acct.Memory.Equity()
        totalEquity += eq

        for _, pos := range acct.Memory.Positions() {
            notional := math.Abs(pos.Notional)
            existingExposure += notional
            if pos.Symbol == plan.Signal.Symbol {
                symbolExposure += notional
            }
            if pos.Direction == plan.Signal.Direction {
                sameDirectionExp += notional
            }
        }
        dailyLossTotal += acct.DailyPnL
    }

    for _, candidate := range plan.Candidates {
        proposedExposure += candidate.ProposedNotional
        if candidate.Symbol == plan.Signal.Symbol {
            symbolExposure += candidate.ProposedNotional
        }
        if candidate.Direction == plan.Signal.Direction {
            sameDirectionExp += candidate.ProposedNotional
        }
    }

    if totalEquity == 0 {
        return &risk.Decision{Allowed: false, Layer: "global", Reason: "total equity is zero"}
    }

    totalExposure := existingExposure + proposedExposure

    // 检查 4 个全局限制
    if totalExposure/totalEquity > g.config.MaxGlobalExposurePct/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global exposure exceeded"}
    }
    if symbolExposure/totalEquity > g.config.MaxSymbolExposure/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: fmt.Sprintf("%s global exposure exceeded", signal.Symbol)}
    }
    if sameDirectionExp/totalEquity > g.config.MaxGlobalSameDirection/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global same-direction exceeded"}
    }
    if dailyLossTotal/totalEquity < -g.config.MaxGlobalDailyLoss/100 {
        return &risk.Decision{Allowed: false, Layer: "global",
            Reason: "global daily loss exceeded"}
    }

    return &risk.Decision{Allowed: true}
}
```

约束细化：
- `GlobalRiskGuard` 必须看到“已有仓位 + 本次候选单”，否则会出现检查通过但下单后超限
- LLM 审查返回 `size_factor` 后，需要对 `DispatchPlan` 应用缩放并**再次执行**全局风控
- 被缩量或移除的 candidate 必须进入 `SignalTrace`，不能只记录最终成交结果

**未来预留**: 如果风控复杂度增长，GlobalRiskGuard 可以零修改拉出来变成独立 Risk Brain sidecar。

## 五、信号审计追踪（SignalTrace）

每笔交易决策的完整链路可追溯：

```go
type SignalTrace struct {
    TraceID    string
    Timestamp  time.Time
    Symbol     string

    // 数据快照
    SnapshotSeq uint64
    Price       float64
    Features    [192]float64

    // 各策略原始信号
    Signals     []strategy.Signal

    // 聚合结果
    Aggregated  strategy.AggregatedSignal

    // 候选单（审查前）
    DraftCandidates []DispatchCandidate

    // 全局风控结果
    GlobalRisk  risk.Decision

    // 审查结果
    Review      *ReviewTrace

    // 各账号处理结果
    AccountResults []AccountTraceResult

    // 被拒绝时记录卡在哪一层
    RejectedStage string  // "per_account_precheck", "global_risk_pre", "review", "global_risk_post"
}

type ReviewTrace struct {
    Requested   bool
    Trigger     string
    Approved    bool
    SizeFactor  float64
    Reason      string
    Latency     time.Duration
}

type AccountTraceResult struct {
    AccountID  string
    RiskResult risk.Decision
    Intent     *execution.OrderIntent
    ExecResult *execution.ExecutionResult
    Latency    time.Duration
}

type TraceStore interface {
    Save(ctx context.Context, trace *SignalTrace) error
    Query(ctx context.Context, filter TraceFilter) ([]SignalTrace, error)
}
```

审计追踪回答这些问题：
- 这笔交易为什么开？→ 哪个 tick 触发？
- 每个策略给了什么信号？→ Signals 数组
- 聚合后的分数是多少？→ Aggregated
- 准备发给哪些账号、计划多大仓位？→ DraftCandidates
- 全局风控通过了吗？→ GlobalRisk
- LLM 是否缩量或拒绝？→ Review
- 每个账号怎么处理的？→ AccountResults
- 下单延迟多少？→ Latency

## 六、Tick Loop（核心循环）

```go
type QuantBrain struct {
    rings        map[string]*SharedRingBuffer
    lastSeenSeq  map[string]uint64
    pool         *strategy.Pool
    aggregator   *strategy.Aggregator
    globalGuard  *GlobalRiskGuard
    router       *AccountRouter
    traceStore   TraceStore
    kernelCaller sidecar.KernelCaller
}

func (q *QuantBrain) Run(ctx context.Context) error {
    marketTicker := time.NewTicker(5 * time.Millisecond)
    housekeepingTicker := time.NewTicker(100 * time.Millisecond)
    defer marketTicker.Stop()
    defer housekeepingTicker.Stop()
    for {
        select {
        case <-ctx.Done():
            return ctx.Err()
        case <-marketTicker.C:
            q.processDirtySnapshots(ctx)
        case <-housekeepingTicker.C:
            q.processAccountOrders(ctx)
        }
    }
}

func (q *QuantBrain) processDirtySnapshots(ctx context.Context) {
    for instID, ring := range q.rings {
        if writeSeq := ring.PeekWriteSeq(); writeSeq == q.lastSeenSeq[instID] {
            continue
        }

        snap, ok := ring.ReadLatest()
        if !ok { continue }
        q.lastSeenSeq[instID] = snap.SeqNum

        // 1. 适配 MarketView
        view := q.adaptSnapshot(instID, snap)

        // 2. 策略池（全局，只算一次）
        signals := q.pool.Compute(view)

        // 3. 信号聚合
        agg := q.aggregator.Aggregate(view, signals, q.buildReviewContext())
        if agg.Direction == strategy.DirectionFlat ||
           agg.Direction == strategy.DirectionHold {
            continue
        }

        // 4. 开始审计追踪
        trace := &SignalTrace{
            TraceID:     generateTraceID(),
            Timestamp:   time.Now(),
            Symbol:      instID,
            SnapshotSeq: snap.SeqNum,
            Price:       snap.CurrentPrice,
            Features:    snap.FeatureVector,
            Signals:     signals,
            Aggregated:  *agg,
        }

        // 5. 先做 per-account 预检查，生成候选单
        plan := q.router.BuildDispatchPlan(agg, snap)
        plan.TraceID = trace.TraceID
        trace.DraftCandidates = append(trace.DraftCandidates, plan.Candidates...)
        if len(plan.Candidates) == 0 {
            trace.RejectedStage = "per_account_precheck"
            q.traceStore.Save(ctx, trace)
            continue
        }

        // 6. 全局风控（审查前）
        preDecision := q.globalGuard.Evaluate(plan)
        trace.GlobalRisk = *preDecision
        if !preDecision.Allowed {
            trace.RejectedStage = "global_risk_pre"
            q.traceStore.Save(ctx, trace)
            continue
        }

        // 7. LLM 审查（条件触发）
        if q.NeedsReview(plan) {
            review := q.RequestReview(ctx, plan)
            trace.Review = &ReviewTrace{
                Requested:  true,
                Trigger:    q.buildReviewReason(plan),
                Approved:   review.Approved,
                SizeFactor: review.SizeFactor,
                Reason:     review.Reason,
            }
            if !review.Approved {
                trace.RejectedStage = "review"
                q.traceStore.Save(ctx, trace)
                continue
            }

            // 审查可能缩量，缩量后必须重跑全局风控
            if review.SizeFactor < 1.0 {
                plan.ApplySizeFactor(review.SizeFactor)
                postDecision := q.globalGuard.Evaluate(plan)
                trace.GlobalRisk = *postDecision
                if !postDecision.Allowed {
                    trace.RejectedStage = "global_risk_post"
                    q.traceStore.Save(ctx, trace)
                    continue
                }
            }
        }

        // 8. 多账号分发
        results := q.router.ExecutePlan(ctx, plan)
        trace.AccountResults = results

        // 9. 保存审计追踪
        q.traceStore.Save(ctx, trace)
    }
}
```

关键约束：
- 行情决策链路由 `WriteSeq` 变化驱动，不是固定 10Hz 轮询
- `100ms` 的 housekeeping loop 只处理止损止盈、状态对账、慢速巡检
- 这样可以保持“快路径 < 5ms”与总览文档中的延迟预算一致

## 八、Account Router

```go
type AccountRouter struct {
    accounts   map[string]*AccountState
    normalizer *InstrumentNormalizer
}

func (r *AccountRouter) BuildDispatchPlan(signal *strategy.AggregatedSignal, snap *MarketSnapshot) *DispatchPlan {
    plan := &DispatchPlan{
        Signal:     signal,
        Snapshot:   snap,
        Candidates: make([]DispatchCandidate, 0, len(r.accounts)),
    }

    for _, acct := range r.accounts {
        if !acct.Config.Enabled || acct.Status != "active" { continue }
        if !acct.isStrategyAllowed(signal.DominantStrategy) { continue }
        if !acct.isInstrumentAllowed(signal.Symbol) { continue }

        // per-account 风控先做预检查，通过后才进入全局风控候选集
        decision := acct.Guard.Evaluate(acct.buildOrderRequest(signal, snap))
        if !decision.Allowed {
            continue
        }

        intent := acct.buildOrderIntent(signal, snap)
        qty := parseQuantity(scaleQuantity(intent.Quantity, acct.Config.WeightFactor))
        plan.Candidates = append(plan.Candidates, DispatchCandidate{
            AccountID:        acct.Config.ID,
            Symbol:           signal.Symbol,
            Direction:        signal.Direction,
            ProposedQty:      qty,
            ProposedNotional: qty * snap.CurrentPrice,
            WeightFactor:     acct.Config.WeightFactor,
            RiskResult:       *decision,
        })
    }
    return plan
}

func (r *AccountRouter) ExecutePlan(ctx context.Context, plan *DispatchPlan) []AccountTraceResult {
    results := make([]AccountTraceResult, 0, len(plan.Candidates))

    for _, candidate := range plan.Candidates {
        acct := r.accounts[candidate.AccountID]
        tr := AccountTraceResult{
            AccountID:  candidate.AccountID,
            RiskResult: candidate.RiskResult,
        }
        start := time.Now()

        intent := acct.buildOrderIntent(plan.Signal, plan.Snapshot)
        intent.Quantity = formatQuantity(candidate.ProposedQty)
        intent.Symbol = r.normalizer.Denormalize(acct.Config.Exchange, plan.Signal.Symbol)
        tr.Intent = &intent

        result, err := acct.Executor.Execute(ctx, intent)
        if err != nil {
            acct.Status = "error"
            acct.LastError = err.Error()
            tr.Latency = time.Since(start)
            results = append(results, tr)
            continue
        }

        acct.Memory.Update(result)
        tr.ExecResult = &result
        tr.Latency = time.Since(start)
        results = append(results, tr)
    }

    return results
}
```

## 十、LLM 审查触发

```go
func (q *QuantBrain) NeedsReview(plan *DispatchPlan) bool {
    ctx := q.buildReviewContext()
    return len(plan.Candidates) >= q.config.LLM.TriggerConcurrent ||
           ctx.LargestPositionPct > q.config.LLM.TriggerPositionPct ||
           ctx.DailyLossPct > q.config.LLM.TriggerDailyLoss ||
           ctx.AnomalyDetected
}

func (q *QuantBrain) RequestReview(ctx context.Context, plan *DispatchPlan) *ReviewDecision {
    req := ReviewRequest{
        TraceID:    plan.TraceID,
        Signal:     *plan.Signal,
        Candidates: plan.Candidates,
        Portfolio:  q.globalPortfolioView(),
        Trigger:    q.buildReviewReason(plan),
    }

    var result protocol.ToolCallResult
    err := q.kernelCaller.CallKernel(ctx, protocol.MethodSpecialistCallTool,
        &protocol.SpecialistToolCallRequest{
            TargetKind: agent.KindCentral,
            ToolName:   "central.review_trade",
            Arguments:  mustJSON(req),
        },
        &result,
    )
    if err != nil || result.IsError {
        // LLM 超时不阻塞，降级为自动决策
        return &ReviewDecision{
            Approved:   true,
            SizeFactor: 1.0,
            Reason:     "llm_timeout_fallback",
        }
    }

    var decision ReviewDecision
    if err := result.DecodeOutput(&decision); err != nil {
        return &ReviewDecision{
            Approved:   true,
            SizeFactor: 1.0,
            Reason:     "llm_decode_fallback",
        }
    }
    decision.SizeFactor = clamp(decision.SizeFactor, 0, 1)
    return &decision
}
```

`ReviewDecision.SizeFactor` 的处理规则：
- `0` = 全拒绝，相当于 `Approved=false`
- `(0,1)` = 缩量执行，先应用到 `DispatchPlan`，再重跑 `GlobalRiskGuard`
- `1` = 原始候选单执行
- 超时/解析失败 = `Approved=true, SizeFactor=1.0`，但必须记录到 `SignalTrace.Review`

## 十一、崩溃恢复

```go
func (q *QuantBrain) recoverState(ctx context.Context) error {
    for _, acct := range q.router.accounts {
        // 1. 从 PostgreSQL 恢复交易记录 → MemoryState
        trades, err := q.store.Trades().OpenByAccount(ctx, acct.Config.ID)
        if err != nil { return err }
        for _, t := range trades {
            acct.Memory.RestorePosition(t)
        }

        // 2. live 账号重连 Private WS → 获取最新持仓
        if acct.Config.Mode == "live" && acct.PrivateWS != nil {
            acct.PrivateWS.Reconnect(ctx)
            // 用交易所返回的持仓数据校正 MemoryState
        }

        // 3. 恢复今日损益
        acct.DailyPnL = q.store.Trades().DailyPnL(ctx, acct.Config.ID, time.Now())
    }
    return nil
}
```
